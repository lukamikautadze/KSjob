${company || ''}

${title || 'No Title'}

${tag || 'General'}

${desc || ''}

CALL: ${phone || 'No Number'} `; feed.appendChild(card); } }); } catch (e) { console.error(e); document.getElementById('loader').innerHTML = "Error loading data. Check CSV link."; } } loadJobs();
